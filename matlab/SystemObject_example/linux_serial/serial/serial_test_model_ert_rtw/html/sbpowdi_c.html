10#include "tmwtypes.h"



11#include <math.h>

13real_T powdi(real_T x, int_T n)



14{



15    real_T powAns;



16    uint_T u;



17



18    /*



19     * First executable statement



20     */



21    powAns = 1;



22    if (n != 0) {



23        if (n < 0) {



24            n = -n;



25            x = 1 / x;



26        }



27        for (u = n;;) {



28            if (u & 01)



29                powAns *= x;



30            if (u >>= 1)



31                x *= x;



32            else



33                break;



34        }



35    }



36    return (powAns);                       /* Last card of powdi */



37}
