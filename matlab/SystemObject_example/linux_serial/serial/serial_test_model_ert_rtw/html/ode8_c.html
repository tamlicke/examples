196void rt_ODEUpdateContinuousStates(RTWSolverInfo *si)



197{



198    time_T    t          = rtsiGetT(si);



199    time_T    tnew       = rtsiGetSolverStopTime(si);



200    time_T    h          = rtsiGetStepSize(si);



201    real_T    *x         = rtsiGetContStates(si);



202    IntgData  *intgData  = rtsiGetSolverData(si);



203    real_T    *deltaY    = intgData->deltaY;



204	real_T    *x0        = intgData->x0;



205	real_T*	  f[NSTAGES];



206	int idx,stagesIdx,statesIdx;



207	double deltaX;



208    



209#ifdef NCSTATES



210    int_T     nXc        = NCSTATES;



211#else



212    int_T     nXc        = rtsiGetNumContStates(si);



213#endif



214



215	f[0]        = intgData->f[0];



216    f[1]        = intgData->f[1];



217    f[2]        = intgData->f[2];



218    f[3]        = intgData->f[3];



219    f[4]        = intgData->f[4];



220    f[5]        = intgData->f[5];



221	f[6]        = intgData->f[6];



222	f[7]        = intgData->f[7];



223	f[8]        = intgData->f[8];



224	f[9]        = intgData->f[9];



225	f[10]       = intgData->f[10];



226	f[11]       = intgData->f[11];



227	f[12]       = intgData->f[12];



228



229    rtsiSetSimTimeStep(si,MINOR_TIME_STEP);



230



231    /* Save the state values at time t in y, we'll use x as ynew. */



232    (void)memset(deltaY, 0, nXc*sizeof(real_T));



233	(void)memcpy(x0, x, nXc*sizeof(real_T));



234



235    for(stagesIdx=0;stagesIdx<NSTAGES;stagesIdx++)



236	{



237		memcpy(x,x0,nXc*sizeof(real_T));



238		



239		for(statesIdx=0;statesIdx<nXc;statesIdx++)



240		{



241			deltaX = 0;



242			for(idx=0;idx<stagesIdx;idx++)



243			{



244				deltaX = deltaX + h*rt_ODE8_A[stagesIdx][idx]*f[idx][statesIdx];



245			}



246			x[statesIdx] = x0[statesIdx] + deltaX;



247		}



248		



249        if(stagesIdx==0)



250        {



251            rtsiSetdX(si, f[stagesIdx]);



252            DERIVATIVES(si);



253        }else



254        {



255            (stagesIdx==NSTAGES-1)? rtsiSetT(si, tnew) : rtsiSetT(si, t + h*rt_ODE8_C[stagesIdx]);



256            rtsiSetdX(si, f[stagesIdx]);



257            OUTPUTS(si,0);



258            DERIVATIVES(si);



259        }



260				



261		for(statesIdx=0;statesIdx<nXc;statesIdx++) 



262		{



263			deltaY[statesIdx]  = deltaY[statesIdx] + h*rt_ODE8_B[stagesIdx]*f[stagesIdx][statesIdx];



264		}



265	}



266	



267	for(statesIdx=0;statesIdx<nXc;statesIdx++) 



268	{



269		x[statesIdx] = x0[statesIdx] + deltaY[statesIdx];



270	}



271



272    PROJECTION(si);



273    REDUCTION(si);



274



275    rtsiSetSimTimeStep(si,MAJOR_TIME_STEP);



276}
