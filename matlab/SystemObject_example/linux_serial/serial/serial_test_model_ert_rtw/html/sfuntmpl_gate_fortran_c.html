122/* Function: mdlInitializeSizes ===============================================



123 * Abstract:



124 *    The sizes information is used by Simulink to determine the S-function



125 *    block's characteristics (number of inputs, outputs, states, etc.).



126 */



127static void mdlInitializeSizes(SimStruct *S)



128{



129    ssSetNumSFcnParams(S, 0);  /* Number of expected parameters */



130    if (ssGetNumSFcnParams(S) != ssGetSFcnParamsCount(S)) {



131        /* Return if number of expected != number of actual parameters */



132        return;



133    }



134



135    ssSetNumContStates(S, 1);  /* how many continuous states? */



136    ssSetNumDiscStates(S, 0);



137



138    if (!ssSetNumInputPorts(S, 1)) return;



139    ssSetInputPortWidth(S, 0, 1);



140    /*



141     * Set direct feedthrough flag (1=yes, 0=no).



142     * A port has direct feedthrough if the input is used in either



143     * the mdlOutputs or mdlGetTimeOfNextVarHit functions.



144     */



145    ssSetInputPortDirectFeedThrough(S, 0, 1);



146



147    if (!ssSetNumOutputPorts(S, 1)) return;



148    ssSetOutputPortWidth(S, 0, 1);



149



150    ssSetNumSampleTimes(S, 1);



151



152    /* 



153     * If your Fortran code uses REAL for the state, input, and/or output 



154     * datatypes, use these DWorks as work areas to downcast continuous 



155     * states from double to REAL before calling your code.  You could



156     * also put the work vectors in hard-coded local (stack) variables.



157     *



158     * For fixed step code, keep a copy of the variables  to be output 



159     * in a DWork vector so the mdlOutputs() function can provide output 



160     * data when needed. You can use as many DWork vectors as you like 



161     * for both input and output (or hard-code local variables).



162     */



163    if(!ssSetNumDWork(   S, 3)) return;



164



165    ssSetDWorkWidth(     S, 0, ssGetOutputPortWidth(S,0));



166    ssSetDWorkDataType(  S, 0, SS_SINGLE); /* use SS_DOUBLE if needed */



167



168    ssSetDWorkWidth(     S, 1, ssGetInputPortWidth(S,0));



169    ssSetDWorkDataType(  S, 1, SS_SINGLE);



170



171    ssSetDWorkWidth(     S, 2, ssGetNumContStates(S));



172    ssSetDWorkDataType(  S, 2, SS_SINGLE);



173



174    ssSetNumNonsampledZCs(S, 0);



175



176    /* Specify the sim state compliance to be same as a built-in block */



177    /* see sfun_simstate.c for example of other possible settings */



178    ssSetSimStateCompliance(S, USE_DEFAULT_SIM_STATE);



179



180    ssSetOptions(S, 0);



181}

186 * Abstract:



187 *    This function is used to specify the sample time(s) for your



188 *    S-function. You must register the same number of sample times as



189 *    specified in ssSetNumSampleTimes.

191static void mdlInitializeSampleTimes(SimStruct *S)



192{



193#ifdef VARIABLE_STEP



194    



195    /* 



196     * For Fortran code with either no states at



197     * all or with continuous states that you want



198     * to support with variable time steps, use



199     * a sample time like this:



200     */



201    ssSetSampleTime(S, 0, CONTINUOUS_SAMPLE_TIME);



202    ssSetOffsetTime(S, 0, 0.0);



203    



204#else



205



206    /* 



207     * If the Fortran code implicitly steps time



208     * at a fixed rate and you don't want to change



209     * the code, you need to use a discrete (fixed



210     * step) sample time, 1 second is chosen below.



211     */



212    ssSetSampleTime(S, 0, 0.01); /* Choose the sample time here if discrete */



213    ssSetOffsetTime(S, 0, 0.0);



214   



215#endif



216    ssSetModelReferenceSampleTimeDefaultInheritance(S);



217}

265/* Function: mdlOutputs =======================================================



266 * Abstract:



267 *    In this function, you compute the outputs of your S-function



268 *    block.  The default datatype for signals in Simulink is double,



269 *    but you can use other intrinsic C datatypes or even custom



270 *    datatypes if you wish.  See Simulink document "Writing S-functions"



271 *    for details on datatype topics.



272 */



273static void mdlOutputs(SimStruct *S, int_T tid)



274{



275



276#ifdef VARIABLE_STEP



277



278    /*



279     *    For Variable Step Code WITH CONTINUOUS STATES



280     *    ---------------------------------------------



281     * For Fortran code that implements continuous states and uses



282     * the mdlDerivatives interface, call your Fortran code's output



283     * routines from here.  If it alters the states, you have to



284     * reset the solver.  Remember, in Simulink the continuous states



285     * must be of type double, so be prepared to copy them to float 



286     * if your Fortran code uses REAL as the datatype for the states.



287     *



288     *                ... or, NO STATES



289     *                -----------------



290     * If your code has no states and you want it to execute in



291     * a continuous model, keep the uPtrs, sampleArgs, y, and



292     * sampleOutput variables and delete x, xf, and nx.  Adjust 



293     * the function call accordingly.



294     */ 



295    InputRealPtrsType uPtrs = ssGetInputPortRealSignalPtrs(S,0);



296    float  *sampleArgs   = (float *) ssGetDWork(S,1);



297    double *y            = ssGetOutputPortRealSignal(S,0);



298    float  *sampleOutput = (float *) ssGetDWork(S,0);



299    double *x            = ssGetContStates(S);



300    float  *xf           = (float *) ssGetDWork(S,2);



301    int     nx           = ssGetNumContStates(S);



302    int k;



303    



304    /* 



305     * If the datatype in the Fortran code is REAL



306     * then you have to downcast the I/O and states from



307     * double to float as copies before sending them 



308     * to your code (or change the Fortran code).



309     */



310



311    for (k=0; k < ssGetDWorkWidth(S,1); k++) {



312        sampleArgs[k] = (float) (*uPtrs[k]);



313    }



314



315    /*



316     * It is recommended to use a DWork vector to 



317     * allocate the space for  the float copy of 



318     * the states (if needed).



319     */



320    for (k=0; k < nx; k++) {



321        xf[k] = (float) x[k];



322    }



323    



324



325    /* ==== Call the Fortran routine (args are pass-by-reference) */



326    



327    /* nameofsub_(sampleArgs, xf, &nx, sampleOutput ); */



328



329   



330    /* 



331     * If needed, convert the float outputs to the 



332     * double (y) output array 



333     */



334    for (k=0; k < ssGetOutputPortWidth(S,0); k++) {



335        y[k] = (double) sampleOutput[k];



336    }



337



338#else



339



340    /* 



341     *    For Fixed Step Code



342     *    -------------------



343     * If the Fortran code implements discrete states (implicitly or



344     * registered with Simulink, it doesn't matter), call the code



345     * from mdlUpdates() and save the output values in a DWork vector.  



346     * The variable step solver may call mdlOutputs() several



347     * times in between calls to mdlUpdate, and you must extract the 



348     * values from the DWork vector and copy them to the block output



349     * variables.



350     *



351     * Be sure that the ssSetDWorkDataType(S,0) declaration in 



352     * mdlInitializeSizes() uses SS_DOUBLE for the datatype when 



353     * this code is active.



354     */



355    



356    double *copyOfOutputs = (double *) ssGetDWork(S, 0);



357    double *y             = ssGetOutputPortRealSignal(S,0);



358    int     k;



359    



360    for (k=0; k < ssGetOutputSignalWidth(S,0); k++ ) {



361        y[k] = copyOfOutputs[k];



362    }



363



364#endif



365



366}

372/* Function: mdlUpdate ======================================================



373 * Abstract:



374 *    This function is called once for every major integration time step.



375 *    Discrete states are typically updated here, but this function is useful



376 *    for performing any tasks that should only take place once per



377 *    integration step.

380{



381



382#ifndef VARIABLE_STEP



383



384    /*

390     * continuous if you call your code from here.



391     */

394    double *y            = ssGetOutputPortRealSignal(S,0);

397

399     * If the datatype in the Fortran code is REAL



400     * then you have to downcast the I/O and states from



401     * double to float as copies before sending them 



402     * to your code (or change the Fortran code).



403     */



404

406        sampleArgs[k] = (float) (*uPtrs[k]);



407    }



408



409



410    /* ==== Call the Fortran routine (args are pass-by-reference) */



411    



412    /* nameofsub_(sampleArgs, sampleOutput ); */



413



414   



415    /* 



416     * If needed, convert the float outputs to the 



417     * double (y) output array 



418     */



419    for (k=0; k < ssGetOutputPortWidth(S,0); k++) {



420        y[k] = (double) sampleOutput[k];



421    }
