31typedef struct IntgData_tag {



32    real_T *y;



33    real_T *f[6];



34} IntgData;

36#ifndef RT_MALLOC



37  /* statically declare data */



38  static real_T   rt_ODE5_Y[NCSTATES];



39  static real_T   rt_ODE5_F[6][NCSTATES];



40  static IntgData rt_ODE5_IntgData = {rt_ODE5_Y,



41                                      {rt_ODE5_F[0],



42                                       rt_ODE5_F[1],



43                                       rt_ODE5_F[2],



44                                       rt_ODE5_F[3],



45                                       rt_ODE5_F[4],



46                                       rt_ODE5_F[5]}};

48  void rt_ODECreateIntegrationData(RTWSolverInfo *si)



49  {



50      rtsiSetSolverData(si,(void *)&rt_ODE5_IntgData);



51      rtsiSetSolverName(si,"ode5");



52  }



53#else



54  /* dynamically allocated data */



55



56  void rt_ODECreateIntegrationData(RTWSolverInfo *si)



57  {



58      IntgData *id = (IntgData *) malloc(sizeof(IntgData));



59      if(id == NULL) {



60          rtsiSetErrorStatus(si, RT_MEMORY_ALLOCATION_ERROR);



61          return;



62      }



63      



64      id->y = (real_T *) malloc(7*rtsiGetNumContStates(si) * sizeof(real_T));



65      if(id->y == NULL) {



66          rtsiSetErrorStatus(si, RT_MEMORY_ALLOCATION_ERROR);



67          return;



68      }



69      id->f[0] = id->y + rtsiGetNumContStates(si);



70      id->f[1] = id->f[0] + rtsiGetNumContStates(si);



71      id->f[2] = id->f[1] + rtsiGetNumContStates(si);



72      id->f[3] = id->f[2] + rtsiGetNumContStates(si);



73      id->f[4] = id->f[3] + rtsiGetNumContStates(si);



74      id->f[5] = id->f[4] + rtsiGetNumContStates(si);



75      



76      rtsiSetSolverData(si, (void *)id);



77      rtsiSetSolverName(si,"ode5");



78  }



79



80  void rt_ODEDestroyIntegrationData(RTWSolverInfo *si)



81  {



82      IntgData *id = rtsiGetSolverData(si);



83      



84      if (id != NULL) {



85          if (id->y != NULL) {



86              free(id->y);



87          }



88          free(id);



89          rtsiSetSolverData(si, NULL);



90      }



91  }

94void rt_ODEUpdateContinuousStates(RTWSolverInfo *si)



95{



96    time_T    t          = rtsiGetT(si);



97    time_T    tnew       = rtsiGetSolverStopTime(si);



98    time_T    h          = rtsiGetStepSize(si);



99    real_T    *x         = rtsiGetContStates(si);



100    IntgData  *intgData  = rtsiGetSolverData(si);



101    real_T    *y         = intgData->y;



102    real_T    *f0        = intgData->f[0];



103    real_T    *f1        = intgData->f[1];



104    real_T    *f2        = intgData->f[2];



105    real_T    *f3        = intgData->f[3];



106    real_T    *f4        = intgData->f[4];



107    real_T    *f5        = intgData->f[5];



108    real_T    hB[6];



109    int_T     i;



110



111#ifdef NCSTATES



112    int_T     nXc        = NCSTATES;



113#else



114    int_T     nXc        = rtsiGetNumContStates(si);



115#endif



116



117    rtsiSetSimTimeStep(si,MINOR_TIME_STEP);



118



119    /* Save the state values at time t in y, we'll use x as ynew. */



120    (void)memcpy(y, x, nXc*sizeof(real_T));



121



122    /* Assumes that rtsiSetT and ModelOutputs are up-to-date */



123    /* f0 = f(t,y) */



124    rtsiSetdX(si, f0);



125    DERIVATIVES(si);



126



127    /* f(:,2) = feval(odefile, t + hA(1), y + f*hB(:,1), args(:)(*)); */



128    hB[0] = h * rt_ODE5_B[0][0];



129    for (i = 0; i < nXc; i++) {



130	x[i] = y[i] + (f0[i]*hB[0]);



131    }



132    rtsiSetT(si, t + h*rt_ODE5_A[0]);



133    rtsiSetdX(si, f1);



134    OUTPUTS(si,0);



135    DERIVATIVES(si);



136



137    /* f(:,3) = feval(odefile, t + hA(2), y + f*hB(:,2), args(:)(*)); */



138    for (i = 0; i <= 1; i++) hB[i] = h * rt_ODE5_B[1][i];



139    for (i = 0; i < nXc; i++) {



140	x[i] = y[i] + (f0[i]*hB[0] + f1[i]*hB[1]);



141    }



142    rtsiSetT(si, t + h*rt_ODE5_A[1]);



143    rtsiSetdX(si, f2);



144    OUTPUTS(si,0);



145    DERIVATIVES(si);



146



147    /* f(:,4) = feval(odefile, t + hA(3), y + f*hB(:,3), args(:)(*)); */



148    for (i = 0; i <= 2; i++) hB[i] = h * rt_ODE5_B[2][i];



149    for (i = 0; i < nXc; i++) {



150	x[i] = y[i] + (f0[i]*hB[0] + f1[i]*hB[1] + f2[i]*hB[2]);



151    }



152    rtsiSetT(si, t + h*rt_ODE5_A[2]);



153    rtsiSetdX(si, f3);



154    OUTPUTS(si,0);



155    DERIVATIVES(si);



156



157    /* f(:,5) = feval(odefile, t + hA(4), y + f*hB(:,4), args(:)(*)); */



158    for (i = 0; i <= 3; i++) hB[i] = h * rt_ODE5_B[3][i];



159    for (i = 0; i < nXc; i++) {



160	x[i] = y[i] + (f0[i]*hB[0] + f1[i]*hB[1] + f2[i]*hB[2] +



161		       f3[i]*hB[3]);



162    }



163    rtsiSetT(si, t + h*rt_ODE5_A[3]);



164    rtsiSetdX(si, f4);



165    OUTPUTS(si,0);



166    DERIVATIVES(si);



167



168    /* f(:,6) = feval(odefile, t + hA(5), y + f*hB(:,5), args(:)(*)); */



169    for (i = 0; i <= 4; i++) hB[i] = h * rt_ODE5_B[4][i];



170    for (i = 0; i < nXc; i++) {



171	x[i] = y[i] + (f0[i]*hB[0] + f1[i]*hB[1] + f2[i]*hB[2] +



172		       f3[i]*hB[3] + f4[i]*hB[4]);



173    }



174    rtsiSetT(si, tnew);



175    rtsiSetdX(si, f5);



176    OUTPUTS(si,0);



177    DERIVATIVES(si);



178



179    /* tnew = t + hA(6);



180       ynew = y + f*hB(:,6); */



181    for (i = 0; i <= 5; i++) hB[i] = h * rt_ODE5_B[5][i];



182    for (i = 0; i < nXc; i++) {



183	x[i] = y[i] + (f0[i]*hB[0] + f1[i]*hB[1] + f2[i]*hB[2] +



184		       f3[i]*hB[3] + f4[i]*hB[4] + f5[i]*hB[5]);



185    }



186



187    PROJECTION(si);



188    REDUCTION(si);



189



190    rtsiSetSimTimeStep(si,MAJOR_TIME_STEP);



191}
