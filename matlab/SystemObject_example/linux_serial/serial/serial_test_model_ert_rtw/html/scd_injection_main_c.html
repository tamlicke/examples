29/* Function: mdlInitializeSizes ===============================================*/



30static void mdlInitializeSizes(SimStruct *S)



31{



32    /* This S-function has 6 parameters:



33     * 1) IOType: It is a string that determines the type of action that will be taken



34     * by this S-function. It mainly depends on the type of linearization IO, 



35     * and for input type IOs, depends on whether or not it is the active input that



36     * signal is going to be injected from.



37     * 2) DataType: The data type that the second input and output ports will have.



38     * 3) Dimensions: The dimensions that second input and output ports will have.



39     * 4) PortDataSize: The number of bytes that the second input port has. This will be



40     * used when the second input needs to be copied over to the output ports.



41     * 5) SampleTime: The sample time that the second input and output ports will share.



42     * 6) SignalIndex: The index at which element of the second input we will operate on,



43     * it will be used when the second input port is not a scalar. "-1" refers to "all".



44     */



45    /* Declare necessary variables */



46    real_T     *dataType;



47    real_T     *dimensions;



48    int_T      *intdimensions;



49    int_T      dimsCounter = 1;



50    DECL_AND_INIT_DIMSINFO(dimsInfo);



51    real_T     *sampleTime;



52    



53    



54    ssSetNumSFcnParams(S, 6);



55    



56    if (ssGetNumSFcnParams(S) != ssGetSFcnParamsCount(S)) {



57        /* Return if number of expected != number of actual parameters */



58        return;



59    }



60    



61    ssSetNumContStates(S, 0);



62    ssSetNumDiscStates(S, 0);



63    ssAllowSignalsWithMoreThan2D(S);



64    if (!ssSetNumInputPorts(S, 2)) return;



65    if (!ssSetNumOutputPorts(S, 2)) return;



66    



67    /* First input is a scalar of type double, coming from the output of



68     * S-function "injection_input"*/



69    ssSetInputPortWidth(S, 0, 1);



70	ssSetInputPortDataType(S, 0, SS_DOUBLE);



71    



72    /* Both input are direct feed through and contiguous */



73    ssSetInputPortDirectFeedThrough(S, 0, 1);



74    ssSetInputPortDirectFeedThrough(S, 1, 1);



75    ssSetInputPortRequiredContiguous(S, 0, 1);



76    ssSetInputPortRequiredContiguous(S, 1, 1);



77    



78    /*********************************************************************/



79    /************************* DATA TYPE *********************************/



80    /*********************************************************************/



81    /* Read the data type from the second parameter and set it to second input



82     * port and both output ports. */ 



83    dataType = mxGetPr(DATATYPE_ARG(S));



84    ssSetInputPortDataType(S, 1, (DTypeId) dataType[0]);



85    ssSetOutputPortDataType(S, 0, (DTypeId) dataType[0]);



86    ssSetOutputPortDataType(S, 1, (DTypeId) dataType[0]);



87    



88    /*********************************************************************/



89    /************************* DIMENSIONS ********************************/



90    /*********************************************************************/



91    /* Read the dimension info from the third parameter and set it to second 



92     * input and both outputs. */



93    dimensions = mxGetPr(DIMENSIONS_ARG(S));



94    dimsInfo.numDims = (int_T) dimensions[0];



95    intdimensions = (int_T*) malloc(dimsInfo.numDims*sizeof(int_T));    



96    dimsInfo.width = 1;



97    for (dimsCounter = 0;dimsCounter<dimsInfo.numDims;dimsCounter++) {



98        intdimensions[dimsCounter] = (int_T) dimensions[dimsCounter+1];



99        dimsInfo.width *= (int_T) dimensions[dimsCounter+1];



100    }



101    dimsInfo.dims = (int_T*) intdimensions;



102    /* Store int dimensions in user data to be able to deallocate */



103    ssSetUserData(S,(void*)intdimensions);



104    ssSetInputPortDimensionInfo(S,1,&dimsInfo);



105    ssSetOutputPortDimensionInfo(S,0,&dimsInfo);



106    ssSetOutputPortDimensionInfo(S,1,&dimsInfo);



107    



108    /*********************************************************************/



109    /************************ SAMPLE TIME ********************************/



110    /*********************************************************************/



111    sampleTime = mxGetPr(SAMPLETIME_ARG(S));



112    ssSetNumSampleTimes(S, PORT_BASED_SAMPLE_TIMES);



113    ssSetInputPortSampleTime(S, 0, INHERITED_SAMPLE_TIME);



114    ssSetInputPortSampleTime(S, 1, sampleTime[0]);



115    ssSetOutputPortSampleTime(S, 0, sampleTime[0]);



116    ssSetOutputPortSampleTime(S, 1, sampleTime[0]);



117    ssSetInputPortOffsetTime(S, 1, sampleTime[1]);



118    ssSetOutputPortOffsetTime(S, 0, sampleTime[1]);



119    ssSetOutputPortOffsetTime(S, 1, sampleTime[1]);



120    



121    /* Initialize DWork to hold the values of:



122	 * 1. Operating point value - generic type



123	 * 2. The action that the S-function should take



124	 * 3. The logging location, whether input or output of the S-function */



125    



126    ssSetNumDWork(S,3);



127    /* Operating point value */



128    ssSetDWorkWidth(S,0,dimsInfo.width);



129    ssSetDWorkDataType(S,0,(DTypeId) dataType[0]);



130	/* The action */



131	ssSetDWorkWidth(S,1,1);



132	ssSetDWorkDataType(S,1,SS_INT8);



133	/* Logging whether input or output */



134	ssSetDWorkWidth(S,2,1);



135	ssSetDWorkDataType(S,2,SS_BOOLEAN);



136



137    ssSetNumRWork(S, 0);



138    ssSetNumIWork(S, 0);



139    ssSetNumPWork(S, 0);



140    ssSetNumModes(S, 0);



141    ssSetNumNonsampledZCs(S, 0);



142    ssSetOptions(S,SS_OPTION_CALL_TERMINATE_ON_EXIT);



143 



144}

153  /* Function: mdlStart =======================================================

155static void mdlStart(SimStruct *S) 



156{



157    /* In this function, we will populate permanent settings in work vector



158     * such as action type, so that we do not do such things multiple times



159     * in mdlOutputs. */



160    



161    /* Declare variables and read parameters */



162    char      insertiontype[80];



163    int       portDataSize = (int) *mxGetPr(PORTDATASIZE_ARG(S));



164    /* Read DWork vectors */



165	int8_T    *action = (int8_T*) ssGetDWork(S,1);



166	boolean_T *logging = (boolean_T*) ssGetDWork(S,2);



167    



168    /* Variables to define possible actions */



169    boolean_T   isDirectPass;       /* Directly pass the input to outputs */



170    boolean_T   isAdd;              /* Add injection signal on the top of input */



171    boolean_T   isReplace;          /* Speak out injection signal ignoring input */



172    boolean_T   isOpen;             /* Open the loop, speak out op. value */



173    



174    mxGetString(IOTYPE_ARG(S),insertiontype,80);



175    



176



177



178



179     



180	 /* Determine the type of action and logging location so that



181	  * we do not have to run these ifs at each time step */



182	 



183    /* Check where to log */



184    logging[0] = ((strcmp(insertiontype,"InputOutput") == 0) ||



185                   (strcmp(insertiontype,"InputOutputInactive") == 0) ||



186                   (strcmp(insertiontype,"InputOutputOpen") == 0) ||



187                   (strcmp(insertiontype,"InputOutputInactiveOpen") == 0));



188	 



189     /* Determine the action that needs to be taken */



190    isDirectPass = ((strcmp(insertiontype,"None") == 0) ||



191                    (strcmp(insertiontype,"InputInactive") == 0) ||



192                    (strcmp(insertiontype,"Output") == 0) ||



193                    (strcmp(insertiontype,"InputOutputInactive") == 0) ||



194                    (strcmp(insertiontype,"OutputInputInactive") == 0) );



195	if (isDirectPass) {



196        action[0] = DIRECT_PASS_ACTION;



197        return;



198	}



199    isAdd = ((strcmp(insertiontype,"Input") == 0) ||



200             (strcmp(insertiontype,"InputOutput") == 0) ||



201             (strcmp(insertiontype,"OutputInput") == 0));



202	if (isAdd) {



203        action[0] = ADD_ACTION;



204		return;



205    }



206



207    isReplace = ((strcmp(insertiontype,"InputOpen") == 0) ||



208                 (strcmp(insertiontype,"InputOutputOpen") == 0) ||



209                 (strcmp(insertiontype,"OutputInputOpen") == 0));



210	if (isReplace) {



211        action[0] = REPLACE_ACTION;



212		return;



213	 }



214



215    isOpen = ((strcmp(insertiontype,"InputInactiveOpen") == 0) ||



216              (strcmp(insertiontype,"OutputOpen") == 0) ||



217              (strcmp(insertiontype,"InputOutputInactiveOpen") == 0) ||



218              (strcmp(insertiontype,"OutputInputInactiveOpen") == 0) ||



219              (strcmp(insertiontype,"Open") == 0) );



220	if (isOpen) {



221        action[0] = OPEN_ACTION;



222		return;



223    }    



224



225  }



226#endif /*  MDL_START */

228/* Function: mdlOutputs =======================================================



229 */



230static void mdlOutputs(SimStruct *S, int_T tid)



231{



232    /* Declare variables and read parameters */



233    int     signalIndex = ((int) *mxGetPr(SIGNALINDEX_ARG(S)))-1; /* Adjust for zero-based indexing */



234    int     portDataSize = (int) *mxGetPr(PORTDATASIZE_ARG(S));



235    



236    /* Read the DWork variables */



237    void      *opvalue = ssGetDWork(S,0);



238    int8_T    *action = (int8_T*) ssGetDWork(S,1);



239    boolean_T *isLogOutput = (boolean_T*) ssGetDWork(S,2);



240    



241    /* Input & Output variables as VOID for copying with MEMCPY */



242    const void        *u = ssGetInputPortSignal(S,1);     /* 2nd input port */



243    void              *y = ssGetOutputPortSignal(S,1);       /* 2nd output port */



244    void              *y_log = ssGetOutputPortSignal(S,0);



245    



246    /* Now read input and output with data type double */



247    real_T        *opvalue_real = (real_T*) ssGetDWork(S,0);



248    const real_T  *u_real = ssGetInputPortRealSignal(S,1);



249    const real_T  *u_input_real = ssGetInputPortRealSignal(S,0);



250    real_T        *y_real = ssGetOutputPortRealSignal(S,1);



251    real_T        *ylog_real = ssGetOutputPortRealSignal(S,0);



252    



253    if (ssGetT(S) == ssGetTStart(S)) {



254        /* At t=0, initialize the op. point value to the initial value of second input port */



255        const void *u = ssGetInputPortSignal(S,1);



256        (void)memcpy(opvalue,u,portDataSize);



257    }



258    



259    /* Initially copy the second input to the both outputs */



260    (void)memcpy(y,u,portDataSize);



261    (void)memcpy(y_log,u,portDataSize);



262    /* If it is DirectPass, there is nothing else to be done: return */



263    if (action[0] == DIRECT_PASS_ACTION) {



264        if (isLogOutput[0])



265            /* Copy all elements of y to y_log */



266            (void)memcpy(y_log,y,portDataSize);



267        return;



268    }



269    if (action[0] == OPEN_ACTION) {



270        /* Speak out the operating point value */



271        (void)memcpy(y,opvalue,portDataSize); 



272        /* Handle the logging if output is to be logged */



273        if (isLogOutput[0])



274            /* Copy all elements of y to y_log */



275            (void)memcpy(y_log,y,portDataSize);



276        return;



277    }



278     



279    /* Take the action */



280    if (action[0] == ADD_ACTION) {



281        /* Add the injection signal on the top of input */



282        y_real[signalIndex] = u_real[signalIndex]+u_input_real[0];



283        if (isLogOutput[0])



284            /* Copy all elements of y to y_log */



285            (void)memcpy(y_log,y,portDataSize);



286        return;



287    }



288    if (action[0] == REPLACE_ACTION) {



289        /* Speak out the operating point value */



290        (void)memcpy(y,opvalue,portDataSize);



291        /* Speak out the injection signal ignoring the input at the specified index */



292        y_real[signalIndex] = opvalue_real[signalIndex]+u_input_real[0];



293        /* Handle the logging and return */



294        if (isLogOutput[0])



295            /* Copy all elements of y to y_log */



296            (void)memcpy(y_log,y,portDataSize);



297        return;



298    }



299}



300static void mdlTerminate(SimStruct *S)



301{



302    /* Deallocate int dimensions stored in user data*/



303    int_T *intdimensions = ssGetUserData(S);



304    free(intdimensions); /*sbcheck::termok*/



305}
