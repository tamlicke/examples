81  void rt_ODECreateIntegrationData(RTWSolverInfo *si)



82  {



83      int_T nx    = rtsiGetNumContStates(si);



84      int_T vsize = nx * sizeof(real_T);



85      int_T msize = nx * vsize;



86      int_T size  = (6+MAXORDER)*vsize + 2*msize + nx*sizeof(int_T); 



87



88      IntgData *id = (IntgData *) malloc(sizeof(IntgData));



89      if(id == NULL) {



90          rtsiSetErrorStatus(si, RT_MEMORY_ALLOCATION_ERROR);



91          return;



92      }



93      



94      id->x0 = (real_T *) malloc(size);



95      if(id->x0 == NULL) {



96          rtsiSetErrorStatus(si, RT_MEMORY_ALLOCATION_ERROR);



97          return;



98      }



99      id->f0      = id->x0      + nx;



100      id->x1start = id->f0      + nx;



101      id->f1      = id->x1start + nx;



102      id->Delta   = id->f1      + nx;



103      id->E       = id->Delta   + nx;



104      id->fac     = id->E       + MAXORDER * nx;



105      id->DFDX    = id->fac     + nx;



106      id->W       = id->DFDX    + nx * nx;



107      id->pivots  = (int32_T *) (id->W + nx * nx);



108



109      { /* Initialize */



110	  real_T SQRT_EPS = 1.5e-8;   /* sqrt(utGetEps()); */



111	  int_T  i;



112	  for (i = 0; i < nx; i++) {



113	      id->fac[i] = SQRT_EPS;



114	  } 



115      }



116



117      rtsiSetSolverData(si, (void *)id);



118      rtsiSetSolverName(si,"ode14x");



119  }

136/* Simplified version of numjac.cpp, for use with RTW. */



137void local_numjac(RTWSolverInfo   *si,



138		  real_T          *y,



139		  const real_T    *Fty,



140		  real_T          *fac,



141		  real_T          *dFdy)



142{



143    /* constants */



144    real_T THRESH = 1e-6;



145    real_T EPS    = 2.2e-16;  /* utGetEps(); */



146    real_T BL     = pow(EPS, 0.75);



147    real_T BU     = pow(EPS, 0.25);



148    real_T FACMIN = pow(EPS, 0.78);



149    real_T FACMAX = 0.1;



150



151#ifdef NCSTATES



152    int_T     nx = NCSTATES;



153#else



154    int_T     nx = rtsiGetNumContStates(si);



155#endif



156



157    real_T    *x = rtsiGetContStates(si);



158    real_T    del;



159    real_T    difmax;



160    real_T    FdelRowmax;



161    real_T    temp;



162    real_T    Fdiff;



163    real_T    maybe;



164    real_T    xscale;



165    real_T    fscale;



166    real_T    *p;



167    int_T     rowmax;



168    int_T     i,j;



169



170    if (x != y) (void)memcpy(x,y,nx*sizeof(real_T));



171



172    for (p = dFdy, j = 0; j < nx; j++, p += nx) {



173



174        /* Select an increment del for a difference approximation to



175           column j of dFdy.  The vector fac accounts for experience



176           gained in previous calls to numjac. */



177        xscale = fabs(x[j]);



178        if (xscale < THRESH) xscale = THRESH;



179	temp = (x[j] + fac[j]*xscale); 



180        del  = temp  - y[j];



181        while (del == 0.0) {



182            if (fac[j] < FACMAX) {



183                fac[j] *= 100.0;



184                if (fac[j] > FACMAX) fac[j] = FACMAX;



185		temp = (x[j] + fac[j]*xscale); 



186                del  = temp  - x[j];



187            } else {



188                del = THRESH; /* thresh is nonzero */



189                break;



190            }



191        }



192        /* Keep del pointing into region. */



193        if (Fty[j] >= 0.0) del = fabs(del);



194        else del = -fabs(del);



195



196        /* Form a difference approximation to column j of dFdy. */



197        temp = x[j];



198        x[j] += del;



199



200	rtsiSetdX(si,p);



201	OUTPUTS(si,0);



202	DERIVATIVES(si);



203



204        x[j] = temp;



205        difmax = 0.0;



206        rowmax = 0;



207        FdelRowmax = p[0];



208        temp = 1.0 / del;



209        for (i = 0; i < nx; i++) {



210            Fdiff = p[i] - Fty[i];



211            maybe = fabs(Fdiff);



212            if (maybe > difmax) {



213                difmax = maybe;



214                rowmax = i;



215                FdelRowmax = p[i];



216            }



217            p[i] = temp * Fdiff;



218        }



219



220        /* Adjust fac for next call to numjac. */



221        if (((FdelRowmax != 0.0) && (Fty[rowmax] != 0.0)) || (difmax == 0.0)) {



222            fscale = fabs(FdelRowmax);



223            if (fscale < fabs(Fty[rowmax])) fscale = fabs(Fty[rowmax]);



224



225	    if (difmax <= BL*fscale) {



226                /* The difference is small, so increase the increment. */



227                fac[j] *= 10.0;



228                if (fac[j] > FACMAX) fac[j] = FACMAX;



229



230            } else if (difmax > BU*fscale) {



231                /* The difference is large, so reduce the increment. */



232                fac[j] *= 0.1;



233                if (fac[j] < FACMIN) fac[j] = FACMIN;



234



235            }



236        }



237    }



238



239} /* end local_numjac */

242void rt_ODEUpdateContinuousStates(RTWSolverInfo *si)



243{



244    time_T    t0         = rtsiGetT(si);



245    time_T    t1         = t0;



246    time_T    h          = rtsiGetStepSize(si);



247    real_T    *x1        = rtsiGetContStates(si);



248    int_T     order      = rtsiGetSolverExtrapolationOrder(si);



249    int_T     numIter    = rtsiGetSolverNumberNewtonIterations(si);



250



251    IntgData  *id        = rtsiGetSolverData(si);



252    real_T    *x0        = id->x0;



253    real_T    *f0        = id->f0;



254    real_T    *x1start   = id->x1start;



255    real_T    *f1        = id->f1;



256    real_T    *Delta     = id->Delta;



257    real_T    *E         = id->E;



258    real_T    *fac       = id->fac;



259    real_T    *dfdx      = id->DFDX;



260    real_T    *W         = id->W;



261    int_T     *pivots    = id->pivots;



262    int_T     *N         = &(rt_ODE14x_N[0]); 



263    int_T     i,j,k,iter;



264



265#ifdef NCSTATES



266    int_T     nx        = NCSTATES;



267#else



268    int_T     nx        = rtsiGetNumContStates(si);



269#endif



270



271    rtsiSetSimTimeStep(si,MINOR_TIME_STEP);



272



273    /* Save the state values at time t in y, we'll use x as ynew. */



274    (void)memcpy(x0, x1, nx*sizeof(real_T));



275



276    /* Assumes that rtsiSetT and ModelOutputs are up-to-date */



277    /* f0 = f(t,y) */



278    rtsiSetdX(si, f0);



279    DERIVATIVES(si);



280



281    /* Compute the Jacobian */



282    local_numjac(si,x0,f0,fac,dfdx);



283



284    for (j = 0; j < order; j++) {



285	



286	real_T *p;



287	real_T hN = h / N[j];



288	



289	/* Get the iteration matrix and solution at t0 */



290



291	/* [L,U] = lu(I - hN*J) */



292        (void) memcpy(W, dfdx, nx*nx*sizeof(real_T));



293        for (p = W, i = 0; i < nx*nx; i++, p++) *p *= (-hN);



294        for (p = W, i = 0; i < nx; i++, p += (nx+1)) *p += 1.0;



295	rt_lu_real(W,nx,pivots);



296



297	/* First Newton's iteration at t0. */



298	/* rhs = hN*f0  */



299	for (i = 0; i < nx; i++) Delta[i] = hN*f0[i];



300	/* Delta = (U \ (L \ rhs)) */



301	rt_ForwardSubstitutionRR_Dbl(W,Delta,f1,nx,1,pivots,1);



302	rt_BackwardSubstitutionRR_Dbl(W+nx*nx-1,f1+nx-1,Delta,nx,1,0);



303	/* ytmp = y0 + Delta */ 



304	(void)memcpy(x1, x0, nx*sizeof(real_T));



305	for (i = 0; i < nx; i++) x1[i] += Delta[i];



306



307	/* Additional Newton's iterations, if desired. 



308	   for iter = 2:NewtIter



309	     rhs = (yn - ytmp) + hN*feval(odefun,tn,ytmp,extraArgs{:});



310	     Delta = ( U \ ( L \ rhs ) );



311	     ytmp = ytmp + Delta;



312	   end  



313	*/



314	rtsiSetT(si, t0);



315	rtsiSetdX(si, f1);



316	for (iter = 1; iter < numIter; iter++) {



317



318	    OUTPUTS(si,0);



319	    DERIVATIVES(si);



320



321	    for (i = 0; i < nx; i++) Delta[i] = (x0[i]-x1[i]) + hN*f1[i];



322



323	    rt_ForwardSubstitutionRR_Dbl(W,Delta,f1,nx,1,pivots,1);



324	    rt_BackwardSubstitutionRR_Dbl(W+nx*nx-1,f1+nx-1,Delta,nx,1,0);



325



326	    for (i = 0; i < nx; i++) x1[i] += Delta[i];



327	}



328



329	/* Subintegration of N(j) steps for extrapolation 



330	   ttmp = t0;



331	   for i = 2:N(j)



332	     ttmp = ttmp + hN



333	     ytmp0 = ytmp;



334	     for iter = 1:NewtIter



335               rhs = (ytmp0 - ytmp) + hN*feval(odefun,ttmp,ytmp,extraArgs{:});



336	       Delta = ( U \ ( L \ rhs ) );



337	       ytmp = ytmp + Delta;



338	     end



339	   end 



340	*/



341	for (k = 1; k < N[j]; k++) {



342	    t1 = t0 + k*hN;



343	    (void)memcpy(x1start, x1, nx*sizeof(real_T));



344	    rtsiSetT(si, t1);



345	    rtsiSetdX(si, f1);



346	    for (iter = 0; iter < numIter; iter++) {



347



348		OUTPUTS(si,0);



349		DERIVATIVES(si);



350



351		if (iter == 0) {



352		    for (i = 0; i < nx; i++) Delta[i] = hN*f1[i];



353		} else {



354		    for (i = 0; i < nx; i++) Delta[i] = (x1start[i]-x1[i]) + hN*f1[i];



355		}



356



357		/* Modeled after rt_matdivrr_dbl.c, use f1 as a temp storage */



358		rt_ForwardSubstitutionRR_Dbl(W,Delta,f1,nx,1,pivots,1);



359		rt_BackwardSubstitutionRR_Dbl(W+nx*nx-1,f1+nx-1,Delta,nx,1,0);



360



361		for (i = 0; i < nx; i++) x1[i] += Delta[i];



362	    }   



363	}



364



365	/* Extrapolate to order j



366	   E(:,j) = ytmp



367	   for k = j:-1:2



368             coef = N(k-1)/(N(j) - N(k-1))



369             E(:,k-1) = E(:,k) + coef*( E(:,k) - E(:,k-1) )



370	   end 



371	*/



372	(void)memcpy( &(E[nx*j]), x1, nx*sizeof(real_T));



373	for (k = j; k > 0; k--) {



374	    real_T coef = (real_T)(N[k-1]) / (N[j]-N[k-1]);



375



376	    for (i = 0; i < nx; i++) {



377		x1[i] = E[nx*k+i] + coef*(E[nx*k+i] - E[nx*(k-1)+i]);



378	    }



379



380	    (void)memcpy( &(E[nx*(k-1)]), x1, nx*sizeof(real_T));



381	}



382    }



383



384    /* Extrapolated solution



385       x1 = E(:,1);



386    */



387    (void)memcpy(x1, E, nx*sizeof(real_T));



388



389    /* t1 = t0 + h; */



390    rtsiSetT(si,rtsiGetSolverStopTime(si));



391



392    PROJECTION(si);



393    REDUCTION(si);



394



395    rtsiSetSimTimeStep(si,MAJOR_TIME_STEP);



396}
